  <script>

    const buttons = document.querySelectorAll('.tab-btn');

    const panels = document.querySelectorAll('.panel');

    const searchInput = document.getElementById('searchInput');

    const sortSelect = document.getElementById('sortSelect');

    const filterSelect = document.getElementById('filterSelect');


    buttons.forEach(button => {

      button.addEventListener('click', () => {

        buttons.forEach(btn => btn.classList.remove('active'));

        panels.forEach(panel => panel.classList.remove('active'));

        button.classList.add('active');

        document.getElementById(button.dataset.tab).classList.add('active');

      });

    });


    const tierScore = {

      HT5: 1, LT5: 2,

      HT4: 3, LT4: 4,

      HT3: 5, LT3: 6,

      HT2: 7, LT2: 8,

      HT1: 9, LT1: 10

    };


    let allResults = [];


    function initial(name) {

      return (name || '?').trim().charAt(0).toUpperCase();

    }


    function isTier1(rank) {

      return rank === 'HT1' || rank === 'LT1';

    }


    function score(rank) {

      return tierScore[rank] || 0;

    }


    function renderEmpty(el, text) {

      el.innerHTML = `<div class="empty">${text}</div>`;

    }


    function playerAvatar(player) {

      return `<div class="avatar">${initial(player.username)}</div>`;

    }


    function applyFilters(players) {

      const search = searchInput.value.trim().toLowerCase();

      const sort = sortSelect.value;

      const filter = filterSelect.value;


      let filtered = players.filter(player => {

        const matchesSearch = !search || (player.username || '').toLowerCase().includes(search);

        if (!matchesSearch) return false;


        if (filter === 'tier1') return isTier1(player.smp) || isTier1(player.mace);

        if (filter === 'dual') return !!player.smp && !!player.mace;

        return true;

      });


      filtered.sort((a, b) => {

        if (sort === 'name') return (a.username || '').localeCompare(b.username || '');


        if (sort === 'recent') {

          return (new Date(b.updated_at || 0)).getTime() - (new Date(a.updated_at || 0)).getTime();

        }


        const aTier1 = (isTier1(a.smp) ? 1 : 0) + (isTier1(a.mace) ? 1 : 0);

        const bTier1 = (isTier1(b.smp) ? 1 : 0) + (isTier1(b.mace) ? 1 : 0);

        if (bTier1 !== aTier1) return bTier1 - aTier1;


        const aCombined = score(a.smp) + score(a.mace);

        const bCombined = score(b.smp) + score(b.mace);

        if (bCombined !== aCombined) return bCombined - aCombined;


        return (a.username || '').localeCompare(b.username || '');

      });


      return filtered;

    }


    function buildCombinedPlayers(results) {

      const map = new Map();


      results.forEach(result => {

        const key = String(result.discord_id || result.username || '').toLowerCase();

        if (!key) return;


        if (!map.has(key)) {

          map.set(key, {

            discord_id: result.discord_id || '',

            username: result.username || 'Unknown',

            smp: null,

            mace: null,

            updated_at: result.created_at || new Date().toISOString(),

          });

        }


        const player = map.get(key);

        const mode = String(result.mode || '').toLowerCase();

        if (mode === 'smp') player.smp = result.rank_earned || null;

        if (mode === 'mace') player.mace = result.rank_earned || null;

        player.updated_at = result.created_at || new Date().toISOString();

      });


      return [...map.values()];

    }


    function renderStats(players, results) {

      document.getElementById('stat-tests').textContent = results.length;

      document.getElementById('stat-smp').textContent = players.filter(p => p.smp).length;

      document.getElementById('stat-mace').textContent = players.filter(p => p.mace).length;

      document.getElementById('stat-dual').textContent = players.filter(p => p.smp && p.mace).length;

    }


    function renderLeaderboard(players) {

      const leaderboard = document.getElementById('leaderboardList');

      if (!players.length) return renderEmpty(leaderboard, 'No ranked players yet. Submit a test from the bot to populate the leaderboard.');


      leaderboard.innerHTML = players.map((player, index) => {

        const tier1Count = (isTier1(player.smp) ? 1 : 0) + (isTier1(player.mace) ? 1 : 0);

        const meta = tier1Count === 2 ? 'Tier 1 in both modes' : tier1Count === 1 ? 'Only one tier 1 mode' : 'Ranked player';

        return `

          <div class="entry">

            <div class="entry-left">

              <div class="place">#${index + 1}</div>

              ${playerAvatar(player)}

              <div>

                <div class="name">${player.username}</div>

                <div class="meta">${meta}</div>

              </div>

            </div>

            <div class="ranks">

              <span class="small-rank">SMP: ${player.smp || 'Unranked'}</span>

              <span class="small-rank">Mace: ${player.mace || 'Unranked'}</span>

            </div>

          </div>

        `;

      }).join('');

    }


    function renderModeList(elId, results, mode) {

      const list = document.getElementById(elId);

      const search = searchInput.value.trim().toLowerCase();


      const filtered = results

        .filter(r => String(r.mode || '').toLowerCase() === mode)

        .filter(r => !search || String(r.username || '').toLowerCase().includes(search))

        .sort((a, b) => {

          if (sortSelect.value === 'name') return String(a.username || '').localeCompare(String(b.username || ''));

          if (sortSelect.value === 'recent') return (new Date(b.created_at || 0)).getTime() - (new Date(a.created_at || 0)).getTime();

          return score(b.rank_earned) - score(a.rank_earned);

        });


      if (!filtered.length) return renderEmpty(list, `No ${mode.toUpperCase()} tests yet.`);


      list.innerHTML = filtered.map(result => `

        <div class="list-item">

          <span>${result.username || 'Unknown'} <span style="color: var(--muted)">• ${result.region || 'No Region'}</span></span>

          <strong>${result.rank_earned || 'Unranked'}</strong>

        </div>

      `).join('');

    }


    async function loadResults() {

      try {

        const res = await fetch('/api/test-result?ts=' + Date.now());

        const data = await res.json();

        allResults = Array.isArray(data) ? data : [];

        renderAll();

      } catch (err) {

        console.error(err);

        renderEmpty(document.getElementById('leaderboardList'), 'Could not load results right now.');

        renderEmpty(document.getElementById('smpList'), 'Could not load SMP results.');

        renderEmpty(document.getElementById('maceList'), 'Could not load Mace results.');

      }

    }


    function renderAll() {

      const combinedPlayers = buildCombinedPlayers(allResults);

      const filteredPlayers = applyFilters(combinedPlayers);

      renderStats(combinedPlayers, allResults);

      renderLeaderboard(filteredPlayers);

      renderModeList('smpList', allResults, 'smp');

      renderModeList('maceList', allResults, 'mace');

    }


    searchInput.addEventListener('input', renderAll);

    sortSelect.addEventListener('change', renderAll);

    filterSelect.addEventListener('change', renderAll);


    loadResults();

    setInterval(loadResults, 5000);
