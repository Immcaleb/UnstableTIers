<script>

const tierScore={
HT5:1,LT5:2,
HT4:3,LT4:4,
HT3:5,LT3:6,
HT2:7,LT2:8,
HT1:9,LT1:10
}

function score(t){return tierScore[t]||0}

function skin(name){
return `https://mc-heads.net/avatar/${encodeURIComponent(name)}/100`
}

let results=[]

async function load(){
const res=await fetch("/api/test-result?"+Date.now())
results=await res.json()
render()
}

function combine(){
const map={}

results.forEach(r=>{
const name=(r.username||"unknown").toLowerCase()

if(!map[name]){
map[name]={
username:r.username,
smp:null,
mace:null,
updated:r.created_at
}
}

if(r.mode==="smp") map[name].smp=r.rank_earned
if(r.mode==="mace") map[name].mace=r.rank_earned
})

return Object.values(map)
}

function render(){

const players=combine()

document.getElementById("stat-tests").textContent=results.length
document.getElementById("stat-smp").textContent=players.filter(p=>p.smp).length
document.getElementById("stat-mace").textContent=players.filter(p=>p.mace).length
document.getElementById("stat-dual").textContent=players.filter(p=>p.smp&&p.mace).length

const board=document.getElementById("leaderboardList")

if(!players.length){
board.innerHTML=`<div class="empty">No players yet</div>`
return
}

players.sort((a,b)=>score(b.smp)+score(b.mace)-score(a.smp)-score(a.mace))

board.innerHTML=players.map((p,i)=>`
<div class="entry">
<div class="entry-left">
<div class="place">#${i+1}</div>
<img class="avatar" src="${skin(p.username)}">
<div class="name">${p.username}</div>
</div>
<div class="ranks">
<div class="rank">SMP: ${p.smp||"Unranked"}</div>
<div class="rank">Mace: ${p.mace||"Unranked"}</div>
</div>
</div>
`).join("")

const smp=document.getElementById("smpList")
smp.innerHTML=results.filter(r=>r.mode==="smp").map(r=>`
<div class="list-item">
<span>${r.username}</span>
<strong>${r.rank_earned}</strong>
</div>
`).join("")

const mace=document.getElementById("maceList")
mace.innerHTML=results.filter(r=>r.mode==="mace").map(r=>`
<div class="list-item">
<span>${r.username}</span>
<strong>${r.rank_earned}</strong>
</div>
`).join("")

}

load()
setInterval(load,5000)

document.querySelectorAll(".tab-btn").forEach(b=>{
b.onclick=()=>{
document.querySelectorAll(".tab-btn").forEach(x=>x.classList.remove("active"))
document.querySelectorAll(".panel").forEach(x=>x.classList.remove("active"))
b.classList.add("active")
document.getElementById(b.dataset.tab).classList.add("active")
}
})

</script>
